#include "Arduino.h"
#include "diffSteer.h"
#include <math.h>

diffSteer::diffSteer(float wheelBase_fm, float wheelBase_mr, float trackWidth_f, float trackWidth_m, float trackWidth_r)
{
  _wheelBase_fm = wheelBase_fm;
  _wheelBase_mr = wheelBase_mr;
  _trackWidth_f = trackWidth_f;
  _trackWidth_m = trackWidth_m;
  _trackWidth_r = trackWidth_r;
  
}

void diffSteer::calcSpeed(float speed, float angle, int &fl, int &fr, int &ml, int &mr, int &rl, int &rr)
{
  if (abs(angle - 90.0) < 1.0)
  {
    fl = fr = ml = mr = rl = rr = (int)speed;
    return;
  }

  angle = constrain(angle, 30, 150);

  float delta = radians(angle - 90.0);  
  float tanDelta = tan(delta);

  float R = _wheelBase / tanDelta;

  float r_fl = sqrt(pow(R + _trackWidth_f / 2.0f, 2) + pow(_wheelBase / 2.0f, 2));
  float r_fr = sqrt(pow(R - _trackWidth_f / 2.0f, 2) + pow(_wheelBase / 2.0f, 2));
  float r_ml = sqrt(pow(R + _trackWidth_m / 2.0f, 2) + pow(_wheelBase / 2.0f, 2));
  float r_mr = sqrt(pow(R - _trackWidth_m / 2.0f, 2) + pow(_wheelBase / 2.0f, 2));
  float r_rl = sqrt(pow(R + _trackWidth_r / 2.0f, 2) + pow(_wheelBase / 2.0f, 2));
  float r_rr = sqrt(pow(R - _trackWidth_r / 2.0f, 2) + pow(_wheelBase / 2.0f, 2));

  //Serial.print("r_fl="); Serial.println(r_fl);
  //Serial.print("r_fr="); Serial.println(r_fr);
  //Serial.print("r_ml="); Serial.println(r_ml);
  //Serial.print("r_mr="); Serial.println(r_mr);
  //Serial.print("r_rl="); Serial.println(r_rl);
  //Serial.print("r_rr="); Serial.println(r_rr);


  float r_max = max(r_fl, max(r_fr, max(r_rl, r_rr)));

  float v_fl = speed * r_fl / r_max;
  float v_fr = speed * r_fr / r_max;
  float v_rl = speed * r_rl / r_max;
  float v_rr = speed * r_rr / r_max;

  // average front and rear on each side
  float v_left  = (v_fl + v_rl) / 2.0f;
  float v_right = (v_fr + v_rr) / 2.0f;

  fl = rl = (int)constrain(v_left,  0, 255);
  fr = rr = (int)constrain(v_right, 0, 255);

  ml = (int)constrain(v_left  * r_ml / r_fl, 0, 255);
  mr = (int)constrain(v_right * r_mr / r_fr, 0, 255);
}

void diffSteer::calcSteerAngle(float angle, int &s_fl, int &s_fr, int &s_rl, int &s_rr)
{
  if (abs(angle - 90.0) < 1.0)
  {
    s_fl = s_fr = s_rl = s_rr = 0;
    return;
  }

  angle = constrain(angle, 30, 150);

  float delta    = radians(angle - 90.0); 
  float tanDelta = tan(delta);

  float deltaL = atan(_wheelBase * tanDelta / (_wheelBase + 0.5f * _trackWidth_f * tanDelta));
  float deltaR = atan(_wheelBase * tanDelta / (_wheelBase - 0.5f * _trackWidth_f * tanDelta));

  _deltaL = deltaL;
  _deltaR = deltaR;

  s_fl = (int)degrees(deltaL);
  s_fr = (int)degrees(deltaR);
  s_rl = -(int)degrees(deltaL);
  s_rr = -(int)degrees(deltaR);
}
